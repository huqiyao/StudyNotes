## 快速排序

[小灰图解](https://www.sohu.com/a/246785807_684445)

**平均时间复杂度O(nlogn)，最坏O(n<sup>2</sup>)**

最坏情况(已是逆序，要排成顺序)：基准元素不选第一个，随机选个

1. **挖坑法**

```c
int partition(int arr[], int startIndex, int endIndex){
    int left = startIndex;
    int right = endIndex;
    //取第一个元素为基准元素
    int pivot = arr[startIndex];
    //同时第一个坑在其所在位置
    int index = startIndex;
    //此轮分治结束标识是两指针重合
    while(left <= right){ // <= !!!
        //右指针往左移动
        while(left <= right){
            if(arr[right] < pivot){
                //填坑并产生新坑
                arr[index] = arr[right];
                index = right;
                left++;
                break;
            }
            right--;
        }
        //左指针往右移动
        while(left <= right){
            if(arr[left] > pivot){
                arr[index] = arr[left];
                index = left;
                right--;
                break;
            }
            left ++;
        }
    }
    arr[index] = pivot;
    return index;
}
void quickSort(int arr[], int startIndex, int endIndex){
    //递归结束条件：startIndex >= endIndex // 最后分开的部分只有一个元素时结束
    if(startIndex >= endIndex){
        return;
    }
    int pivot = partition(arr, startIndex, endIndex);
    quickSort(arr, startIndex, pivot-1);
    quickSort(arr, pivot+1, endIndex);
}
int main(){
    int arr[] = {1,8,6,5,3,4};
    int length = sizeof(arr)/sizeof(arr[0]);
    quickSort(arr, 0, length);
    for(int i = 0; i < length; i++){
        printf("%d",arr[i]);
    }
}
```

2. **指针交换法**

```c
int partition(int arr[], int startIndex, int endIndex){
    int pivot = arr[startIndex];
    int left = startIndex;
    int right = endIndex;
    while(left != right){
        //控制右指针往左移
        while(left < right && arr[right] >= pivot){
            right--;
        }
        //控制左指针往右移
        while(left < right && arr[left] <= pivot){
            left++;
        }
        //交换left与right指向的元素
        if(left < right){
            int p = arr[left];
            arr[left] = arr[right];
            arr[right] = p;
        }
    }
    //基准数与left、right指针重合点交换
    int p = arr[right];
    arr[right] = pivot;
    pivot = p;
        
    return right;
}
```

```java
public int findKthLargest(int[] nums, int k){
    k = nums.length-k;
    int l = 0, r = nums.length-1;
    while(l < r){
        int j = partition(nums,l,r);
        if(j==k){
            break;
        }else if(j<k){
            l = j+1;
        }else{
            r = j-1;
        }
    }
    return nums[k]; // 如果k刚好是分界点j，就直接找到了；
    				// 如果，k一直不等于j，就在靠近k这端全排完序
}
```

## 冒泡排序

```c
void bubbleSort(int arr[]){
	int length = sizeof(arr)/sizeof(arr[0]);
    for(int i = 0; i < length - 1; i++){ // 比较length-1轮
    	for(int j = 0; j < length - 1 - i; j++){ //已经冒出的肯定是最大的不用继续比
    		if(arr[j] > arr[j+1]){
    			int temp = arr[j];
    			arr[j] = arr[j+1];
    			arr[j+1] = temp;
    		}
    	}
    }
}
```

## 选择排序

```c
void selectSort(int arr[]){
    int minIndex;
    int length = sizeof(arr)/sizeof(arr[0]);
    for(int i = 0; i < length - 1; i++){
    	minIndex = i;
        for(int j = i+1; j < length; j++){
            if(arr[j] < arr[minIndex]){
                minIndex = j;
            }
        }
        if(minIndex != i){
            int temp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = temp;
        }   
    }
}
```

## 归并排序

 1. **递归实现**
   - 递归的方法来分元素
   - 临时数组保存排好序的数
   - 把临时数组中的数据赋值给原数组

```c
// 合并
int mergeAdd(int arr[], int left, int mid, int right, int *temp){
    int i = left;
    int j = mid + 1;
    int k = left;//临时数组的左指针
    while(i <= mid && j <= right){
        if(arr[i] < arr[j]){
            temp[k++] = arr[i++];
        }
        else{
            temp[k++] = arr[j++];
        }
    }
    while(i <= mid){//右边数组结束，只剩下左边时
        temp[k++] = arr[i++];
    }
    while(j <= right){//左边数组结束，只剩下右边时
        temp[k++] = arr[j++];
    }
    // memcpy():从源内存地址的起始位置开始拷贝若干个字节到目标内存地址中
    // 为什么要“+left”？因为可能不是从首地址开始的，加上更安全
    memcpy(arr + left, temp + left, sizeof(int)*(right - left +1));
}
int mergeSort(int arr[], int left, int right, int *temp){
    int mid = 0;
    if(left < right){
        mid = left + (right - left)/2;
        mergeSort(arr, left, mid, temp); 
        mergeSort(arr, mid+1, right, temp);
        mergeAdd(arr, left, mid, right, temp); // 开始并没有执行，直到形成很多组单个的，同时执行
    }
}
// 测试函数
int main(){
    int arr = {8, 4, 5, 1, 3, 6, 2};
    int length = sizeof(arr)/sizeof(arr[0]);
    int *temp = (int*)malloc(sizeof(int)*length);
    mergeSort(arr, 0, length-1, temp);
    free(temp);
    for(int i = 0; i < length; i++){
        printf("%d", arr[i]);
    }
}
```

 ## 堆排序

+ 一种在堆这种数据结构上使用的选择排序
+ 最好、最坏、平均时间复杂度都是O(nlogn)
+ 不稳定

```c
void adjustHeap(int arr[], int i, int length){
    int temp = arr[i]; //取出当前元素
    for(int k = 2*i+1; k < length; k = 2*k+1){ 
    //k=2*i+1:从i的左子结点开始；k=2*k+1:重新回到下一个节点调整  
        if(k + 1 < length && arr[k] < arr[k+1]){ //如果右子节点大于左子结点，指向右子节点
            k++;
        }
        if(arr[i] < arr[k]){//左子节点大于父节点，则将值赋给父节点
            arr[i] = arr[k];
            i = k;
        }
        else{//如果比父节点小，比右兄弟节点大，则跳出循环
            break;
        }
    }
    arr[i] = temp; //最初元素放进最终位子
}
void sort(int arr[]){
	// 构建最大堆
    for(int i = arr.length/2 - 1; i >= 0; i--){
    	// 从第一个非叶子节点自下而上、自坐而右
        adjustHeap(arr, i, arr.length);
    }
    //2.调整堆结构+交换堆顶元素与末尾元素
    for(int j = arr.length-1;j>0;j--){
    	swap(arr,0,j);//将堆顶元素与末尾元素进行交换
        adjustHeap(arr,0,j);//重新对堆进行调整
    }
}
```
